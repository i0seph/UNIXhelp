.mailrc 파일 보기

줄 처음에 # 문자가 있는 것은 주석문입니다. 

   .mailrc 파일 보기

   set askcc hold keepsave editheaders
   # 윗줄은 네개의 변수를 지정한 것입니다. 초기값으로 이들 변수는 
   # 지정되어 있지 않습니다. 물론 각각의 한개의 변수당 한줄로 
   # 따로 따로 지정할 수도 있습니다.
   set folder=mrtidy
   # 이것은 폴더가 저장될 디렉토리를 지정합니다. 
   # 이렇게 절대 경로를 사용하지 않으며 홈디렉토리를 대상으로 계산됩니다.
   set PAGER=less
   # 편지 내용이 한 화면이 넘어갈 경우에 그 내용을들 볼 때 
   # 유용하게 쓰일 pager 프로그램을 지정한 경우입니다.
   set VISUAL=/usr/local/bin/ue
   # 이것은 ~v 명령으로 문서 편집기를 이용해 보낼 편지를 쓸 때 호출될 
   # 편집기를 지정한 것입니다.
   sign="The Boy in the Bubble"
   # 이것은 ~a 명령으로 서명(signature)을 할 때 덧붙혀질 서명내용을 
   # 지정한 것입니다.
